213  AGX_FORCE_INLINE void ShapeColliders::collideCapsulePlane(
214  const agx::AffineMatrix4x4& capsuleTransform,
215  const agx::Real capsuleRadius,
216  const agx::Real capsuleHeight,
217  const agx::Plane& transformedPlane,
218  const agx::Real normalFlipSign,
219  agxCollide::ShapeCollider::LocalContactPointVector& result )
220 
221  {
222  // World coordinates for capsule points.
223  const agx::Vec3 cap1 = agx::Vec3(0, capsuleHeight * agx::Real(0.5), 0);
224  const agx::Vec3 cap2 = agx::Vec3(0, -capsuleHeight * agx::Real(0.5), 0);
225 
226  const agx::Vec3 p1 = cap1 * capsuleTransform;
227  const agx::Vec3 p2 = cap2 * capsuleTransform;
228 
229  const agx::Vec3 normal = transformedPlane.getNormal();
230 
231  // Get the distance from the point to the plane.
232  // If negative or distance < the radius of the sphere, a collision has occurred.
233  const agx::Real d1 = transformedPlane.signedDistanceToPoint(p1);
234  if (d1 <= capsuleRadius) {
235  // The contact point is the capsule endpoint projected onto the plane.
236  const agx::Real depth = capsuleRadius - d1;
237  const agx::Vec3 point = p1 - normal * (d1 + depth * agx::Real(0.5));
238  ShapeCollider::addContactPoint( result, point, normal * normalFlipSign, depth );
239  }
240 
241  const agx::Real d2 = transformedPlane.signedDistanceToPoint(p2);
242  if (d2 <= capsuleRadius) {
243  // The contact point is the capsule endpoint projected onto the plane.
244  const agx::Real depth = capsuleRadius - d2;
245  const agx::Vec3 point = p2 - normal * (d2 + depth * agx::Real(0.5));
246  ShapeCollider::addContactPoint( result, point, normal * normalFlipSign, depth );
247  }
248  }

252  AGX_FORCE_INLINE void ShapeColliders::collideSphereSphere(
253  const agx::Vec3& position1,
254  const agx::Real radius1,
255  const agx::Vec3& position2,
256  const agx::Real radius2,
257  agxCollide::ShapeCollider::LocalContactPointVector& result)
258  {
259  const agx::Real radiusSum = radius1 + radius2;
260  agx::Vec3 normal = position1 - position2;
261 
262  const agx::Real len2 = normal.length2();
263  if (radiusSum * radiusSum < len2) {
264  // Mathematically no contact.
265  // Make sure we don't lose contacts due to floating point precision.
266  agx::Real epsilon = agx::RealEpsilon * (agx::Real(1.0) + radius1 + radius2);
267  if ((radiusSum + epsilon) * (radiusSum + epsilon) < len2)
268  return; // No contact.
269  else {
270  // Contact which might have been missed due to floating point precision.
271  // We keep it (conservative on collision), but give it depth 0.
272  normal.normalize();
273  ShapeCollider::addContactPoint( result,
274  position2 + normal * radius2, normal, agx::Real(0) );
275  }
276  }
277  else {
278  agx::Real distance;
279  // Calculate the normalized contact normal pointing
280  // from sphere2 towards sphere1.
281  if ( len2 == 0 ) {
282  // The sphere centers are at the exact same position.
283  distance = agx::Real( 0 );
284  normal[2] = 1;
285  }
286  else
287  distance = normal.normalize();
288 
289  const agx::Real depth = radiusSum - distance;
290  if (depth >= agx::Real(0)) {
291  ShapeCollider::addContactPoint( result,
292  position2 + normal * (radius2 - depth * agx::Real(0.5)),
293  normal, depth );
294  }
295  }
296  }

299  AGX_FORCE_INLINE void ShapeColliders::collideSpherePlane(
300  const agx::Vec3& spherePosition,
301  const agx::Real sphereRadius,
302  const agx::Plane& transformedPlane,
303  agxCollide::ShapeCollider::LocalContactPointVector& result)
304  {
305  const agx::Real distance = transformedPlane.signedDistanceToPoint(spherePosition);
306  if (distance > sphereRadius)
307  return;
308  const agx::Vec3 normal = transformedPlane.getNormal();
309  const agx::Real depth = sphereRadius - distance;
310  ShapeCollider::addContactPoint( result, spherePosition +
311  normal * (depth * agx::Real(0.5) - sphereRadius), normal, depth );
312  }
