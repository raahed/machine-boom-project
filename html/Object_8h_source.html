333  #define AGX_MODEL_IMPLEMENTATION(_class, _path) \
334  agx::Model *_class::ClassModel() \
335  { \
336  ScopeLock<ReentrantMutex> lock(agx::getTypeSystemMutex()); \
337  static agx::Model *s_model = nullptr; \
338  \
339  if (!s_model) \
340  agx::createImplementedModel(_path, s_model); \
341  \
342  if (!s_model) \
343  LOGGER_ERROR() << agx::String::format("No model found for path \'%s\'. Make sure AGX environment is properly setup", _path) << LOGGER_ENDL(); \
344  return s_model; \
345  }
