321  AGX_FORCE_INLINE bool agxGeometryQueries::LineSegment::containsPoint(const agx::Vec3& point, const agx::Real epsilon) const
322  {
323  const agx::Real projection = projectPointToInfiniteLine(point);
324  if (projection < 0 || projection * projection > m_dirLength2)
325  return false;
326  const agx::Vec3 projectionPoint = m_point0 + m_dir * projection;
327  const agx::Vec3 dist = point - projectionPoint;
328  return dist.length2() < epsilon * epsilon;
329  }
