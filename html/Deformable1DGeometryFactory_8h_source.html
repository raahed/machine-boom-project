306  template<>
307  inline agxCollide::GeometryRef Deformable1DRadiusGeometryFactory< agxCollide::Sphere >::createGeometry(const agxModel::Deformable1DNode* node)
308  {
309  if ( node == nullptr )
310  return agxCollide::GeometryRef( nullptr );
311 
312  // Maybe input?
313  const agx::Real offset = getRadius();
314 
315  const agx::UInt numSpheres = std::max( agx::UInt( node->getLength() / offset + agx::Real( 0.5 ) ), agx::UInt( 1 ) );
316  const agx::Real dl = node->getLength() / agx::Real( numSpheres );
317  agxCollide::GeometryRef geometry = new agxCollide::Geometry();
318  agx::Vec3 pos( 0, 0, 0 );
319  agx::Vec3 dir = agx::Vec3::Z_AXIS();
320  for ( agx::UInt i = 0; i < numSpheres; ++i ) {
321  geometry->add( new agxCollide::Sphere( getRadius() ), agx::AffineMatrix4x4::translate( pos ) );
322  pos += dl * dir;
323  }
324 
325  return geometry;
326  }

328  template< typename T >
329  agxCollide::GeometryRef Deformable1DRadiusGeometryFactory< T >::createGeometry(const agxModel::Deformable1DNode* node)
330  {
331  if ( node == nullptr )
332  return agxCollide::GeometryRef( nullptr );
333 
334  agxCollide::GeometryRef geometry = new agxCollide::Geometry( new T( getRadius(), node->getLength() ) );
335  geometry->setLocalTransform( agx::AffineMatrix4x4::rotate( agx::Vec3::Y_AXIS(), agx::Vec3::Z_AXIS() ) *
336  agx::AffineMatrix4x4::translate( agx::Vec3::Z_AXIS() * agx::Real( 0.5 ) * node->getLength() ) );
337 
338  return geometry;
339  }
