700  {
701  T tmp[3];
702 
703  // Column 0
704  tmp[0] = INNER_PRODUCT(m, *this, 0, 0);
705  tmp[1] = INNER_PRODUCT(m, *this, 1, 0);
706  tmp[2] = INNER_PRODUCT(m, *this, 2, 0);
707 
708  m_data[0][0] = tmp[0];
709  m_data[1][0] = tmp[1];
710  m_data[2][0] = tmp[2];
711 
712  // Column 1
713  tmp[0] = INNER_PRODUCT(m, *this, 0, 1);
714  tmp[1] = INNER_PRODUCT(m, *this, 1, 1);
715  tmp[2] = INNER_PRODUCT(m, *this, 2, 1);
716 
717  m_data[0][1] = tmp[0];
718  m_data[1][1] = tmp[1];
719  m_data[2][1] = tmp[2];
720 
721  // Column 2
722  tmp[0] = INNER_PRODUCT(m, *this, 0, 2);
723  tmp[1] = INNER_PRODUCT(m, *this, 1, 2);

829  {

854  template <typename T>
855  Matrix3x3T<T> Matrix3x3T<T>::inverse() const
856  {
857  /* Check if diagonal matrix */
858  if (this->isDiagonal())
859  return Matrix3x3T<T>(T(1)/m_data[0][0], 0, 0, 0, T(1)/m_data[1][1], 0, 0, 0, T(1)/m_data[2][2]);
860 
861  /* Check if determinant is large enough to do fast inverse */
862  T determinant = this->determinant();
863 
864  if (determinant > agx::RealEpsilon)
865  return this->fastInverse(determinant);
866 
867  /* Else do robust, but slow inversion */
868  const int next[3] = {1, 2, 0};
869  Matrix3x3T<T> result;
870  Matrix3x3T<T> tmp(*this);
871 
872  bool done[3] = {false, false, false};
873  int pivotColumns[3] = {-1, -1, -1};
874 
875  for (int i = 0; i < 3; i++)
876  {
877  // Select pivot row
878  Vec2i coord = findMaxElement(tmp, done);
879  pivotColumns[(int)coord[0]] = (int)coord[1];
880 
881  // Gauss-Jordan pivoting
882  T *row = &tmp(coord[0], 0);
883  T *resultRow = &result(coord[0], 0);
884 
885  // Scale target row
886  T scale = T(1)/row[coord[1]];
887  for (int j = 0; j < 3; j++)
888  {
889  row[j] *= scale;
890  resultRow[j] *= scale;
891  }
892 
893 
894  // Column operations on other rows
895  for (int r = next[coord[0]]; r != coord[0]; r = next[r])
896  {
897  T *ra = &tmp(r, 0);
898  T *rb = &result(r, 0);
899  T f = ra[coord[1]];
900 
901  for (int j = 0; j < 3; j++)
902  {
903  ra[j] -= f*row[j];
904  rb[j] -= f*resultRow[j];
905  }
906  }
907 
908  done[coord[0]] = true;
909  }
910 
911  // Do final row permutations
912  for (int i = 0; i < 3; i++)
913  {
914  T *ra = &tmp(pivotColumns[i], 0);
915  T *rb = &result(i, 0);
916  memcpy(ra, rb, sizeof(T) * 3);
917  }
918 
919  return tmp;
920  }
