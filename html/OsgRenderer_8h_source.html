185  int remapKey(QKeyEvent* event)
186  {
187  KeyMap::iterator itr = mKeyMap.find(event->key());
188  if (itr == mKeyMap.end())
189  {
190  return int(*(event->text().toLatin1().data()));
191  }
192  else
193  return itr->second;
194  }
