119  AGX_FORCE_INLINE agx::Vec3 Convex::getSupportPoint( const agx::Vec3& supportDirection ) const
120  {
121  if ( m_collisionMeshData->m_vertices.empty() )
122  return agx::Vec3();
123  else {
124  agx::Real maxDist = -agx::RealMax;
125  size_t maxIndex = 0;
126  for (size_t i = 0; i < m_collisionMeshData->m_vertices.size(); ++i) {
127  const agx::Real dist = m_collisionMeshData->m_vertices[i] * supportDirection;
128  if (dist > maxDist) {
129  maxDist = dist;
130  maxIndex = i;
131  }
132  }
133  return m_collisionMeshData->m_vertices[maxIndex];
134  }
135  }
