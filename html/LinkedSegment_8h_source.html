209  T* LinkedSegment::createConstraintWithPrev( const LinkedSegment& prev,
210  const agx::Vec3& localConstraintAxis,
211  const agx::Vec3& localConstraintY,
212  agx::Bool disableCollisions )
213  {
214  agxAssert( m_constraint == nullptr );
215  if ( m_constraint != nullptr )
216  return nullptr;
217 
218  const auto localConstraintX = localConstraintY ^ localConstraintAxis;
219  agx::OrthoMatrix3x3 localRotation = agx::OrthoMatrix3x3( localConstraintX.x(), localConstraintX.y(), localConstraintX.z(),
220  localConstraintY.x(), localConstraintY.y(), localConstraintY.z(),
221  localConstraintAxis.x(), localConstraintAxis.y(), localConstraintAxis.z() );
222 
223  agx::FrameRef prevFrame = new agx::Frame();
224  agx::FrameRef thisFrame = new agx::Frame();
225 
226  prevFrame->setLocalMatrix( agx::AffineMatrix4x4( localRotation,
227  agx::Vec3( 0, 0, prev.getLength() ) ) );
228  thisFrame->setLocalMatrix( agx::AffineMatrix4x4( localRotation, agx::Vec3( 0, 0, 0 ) ) );
229 
230  m_constraint = new T( prev.getRigidBody(), prevFrame, getRigidBody(), thisFrame );
231 
232  if ( disableCollisions )
233  agxUtil::setEnableCollisions( prev.getRigidBody(), getRigidBody(), false );
234 
235  return m_constraint->as<T>();
236  }
