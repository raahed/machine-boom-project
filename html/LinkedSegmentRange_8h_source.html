129  : m_begin( begin ), m_end( m_begin.m_index, false, *m_begin.m_segments ), m_size( 0u )
130  {
131  // It's undefined, but possible, to create a range from begin to 10 * container.size().
132  // This assert alerts the user something's probably wrong in their algorithm.
133  // Note: numSegments == container.size() is valid since it can be the range from
134  // begin to end.
