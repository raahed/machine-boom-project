221  template <typename T>
222  bool ArgumentParser::read1(const std::string& key, T& value, size_t startPosition /*= 0*/,
223  bool findAlsoHandled /*= true*/)
224  {
225  int pos = this->find(key, startPosition, findAlsoHandled);
226  if (pos >= 0 && pos + 1 < (int)m_arguments.size()) {
227  if (!assignStringContent (m_arguments[pos + 1].getName(), value))
228  return false;
229  m_arguments[pos].setHandled( true ); // key handled
230  m_arguments[pos + 1].setHandled( true ); // value handled
231  return true;
232  }
