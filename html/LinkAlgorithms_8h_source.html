162  {
163  // If the link has been deleted our contact event listener shouldn't still be in the simulation!
164 
165  // ^^ But if the link has been removed we will send a match to the internal contact event
166  // listener so that the listener can remove itself.
167  if ( m_link == nullptr )
168  return true;
169 
170  agx::Bool firstIsLink = m_link != nullptr && geometry1->getRigidBody() == m_link->getRigidBody();
171  agx::Bool secondIsLink = m_link != nullptr && geometry2->getRigidBody() == m_link->getRigidBody();
172  return firstIsLink ? matchOther( geometry2 ) : secondIsLink ? matchOther( geometry1 ) : false;
173  }
