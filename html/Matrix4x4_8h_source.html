452  template <typename T>
453  AGX_FORCE_INLINE bool Matrix4x4T<T>::isRigidTransformation() const
454  {
455  bool isRigid = true;
456  // rows have norm 1
457  isRigid &= equivalent(m_data[0][0]*m_data[0][0] + m_data[0][1]*m_data[0][1] + m_data[0][2]*m_data[0][2], T(1), T(1.0e-3));
458  isRigid &= equivalent(m_data[1][0]*m_data[1][0] + m_data[1][1]*m_data[1][1] + m_data[1][2]*m_data[1][2], T(1), T(1.0e-3));
459  isRigid &= equivalent(m_data[2][0]*m_data[2][0] + m_data[2][1]*m_data[2][1] + m_data[2][2]*m_data[2][2], T(1), T(1.0e-3));
460  // columns have norm 1
461  isRigid &= equivalent(m_data[0][0]*m_data[0][0] + m_data[1][0]*m_data[1][0] + m_data[2][0]*m_data[2][0], T(1), T(1.0e-3));
462  isRigid &= equivalent(m_data[0][1]*m_data[0][1] + m_data[1][1]*m_data[1][1] + m_data[2][1]*m_data[2][1], T(1), T(1.0e-3));
463  isRigid &= equivalent(m_data[0][2]*m_data[0][2] + m_data[1][2]*m_data[1][2] + m_data[2][2]*m_data[2][2], T(1), T(1.0e-3));
464  // determinant 1
465  isRigid &= equivalent(m_data[0][0]*m_data[1][1]*m_data[2][2] + m_data[0][1]*m_data[1][2]*m_data[2][0] + m_data[0][2]*m_data[1][0]*m_data[2][1]
466  - m_data[2][0]*m_data[1][1]*m_data[0][2] - m_data[1][0]*m_data[0][1]*m_data[2][2] - m_data[0][0]*m_data[2][1]*m_data[1][2],
467  T(1), T(1.0e-1));
468  // no row scaling
469  isRigid &= (m_data[0][3] == T(0));
470  isRigid &= (m_data[1][3] == T(0));
471  isRigid &= (m_data[2][3] == T(0));
472  // no transform scaling
473  isRigid &= (m_data[3][3] == T(1));
474  return isRigid;
475  }

769  template <typename T>
770  AGX_FORCE_INLINE void Matrix4x4T<T>::mult( const Matrix4x4T<T>& lhs, const Matrix4x4T<T>& rhs )
771  {
772  agx::prefetch<agx::L1>( lhs.m_data );
773  agx::prefetch<agx::L1>( rhs.m_data );
774 
775  if (&lhs==this)
776  {
777  postMult(rhs);
778  return;
779  }
780  if (&rhs==this)
781  {
782  preMult(lhs);
783  return;
784  }
785 
786  m_data[0][0] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 0, 0);
787  m_data[0][1] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 0, 1);
788  m_data[0][2] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 0, 2);
789  m_data[0][3] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 0, 3);
790 
791  m_data[1][0] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 1, 0);
792  m_data[1][1] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 1, 1);
793  m_data[1][2] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 1, 2);
794  m_data[1][3] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 1, 3);
795 
796  m_data[2][0] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 2, 0);
797  m_data[2][1] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 2, 1);
798  m_data[2][2] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 2, 2);
799  m_data[2][3] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 2, 3);
800 
801  m_data[3][0] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 3, 0);
802  m_data[3][1] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 3, 1);
803  m_data[3][2] = AGX_MATRIX4X4_INNER_PRODUCT(lhs, rhs, 3, 2);

808  template <typename T>
809  AGX_FORCE_INLINE void Matrix4x4T<T>::preMult( const Matrix4x4T<T>& other )
810  {
811 
812  // more efficient method just use a T[4] for temporary storage.
813  T t[4];
814  for(size_t col=0; col<4; ++col) {
815  t[0] = AGX_MATRIX4X4_INNER_PRODUCT( other, *this, 0, col );
816  t[1] = AGX_MATRIX4X4_INNER_PRODUCT( other, *this, 1, col );
817  t[2] = AGX_MATRIX4X4_INNER_PRODUCT( other, *this, 2, col );
818  t[3] = AGX_MATRIX4X4_INNER_PRODUCT( other, *this, 3, col );
819  m_data[0][col] = t[0];
820  m_data[1][col] = t[1];
821  m_data[2][col] = t[2];
822  m_data[3][col] = t[3];
823  }
824  }

829

831  T t[4];
832  for(size_t row=0; row<4; ++row)
833  {
834  t[0] = AGX_MATRIX4X4_INNER_PRODUCT( *this, other, row, 0 );

836  t[2] = AGX_MATRIX4X4_INNER_PRODUCT( *this, other, row, 2 );
837  t[3] = AGX_MATRIX4X4_INNER_PRODUCT( *this, other, row, 3 );
838 
839  AGX_MATRIX4X4_SET_ROW(row, t[0], t[1], t[2], t[3] )
840  }
