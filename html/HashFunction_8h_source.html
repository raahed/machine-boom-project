100  {
101  /* http://www.concentric.net/~Ttwang/tech/inthash.htm */
102  key = ~key + (key << 15); // key = (key << 15) - key - 1;
103  key = key ^ (key >> 12);
104  key = key + (key << 2);

106  key = key * 2057; // key = (key + (key << 3)) + (key << 11);
