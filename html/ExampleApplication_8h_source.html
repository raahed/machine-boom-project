169  virtual void operator () (osg::RenderInfo& /* renderInfo */) const
170  {
171  // Explicit support for Real=float agx with Real=double osg
172  agxData::Val<agx::Matrix4x4d> projection(agx::Matrix4x4d(m_mainCamera->getProjectionMatrix().ptr()));
173  agx::Matrix4x4 m = projection.transform<agx::Matrix4x4 > ();
174 
175  if (m != m_glCamera->getProjectionMatrix())
176  m_glCamera->setProjectionMatrix(m);
177 
178  agxData::Val<agx::Matrix4x4d> view(agx::Matrix4x4d(m_mainCamera->getViewMatrix().ptr()));
179  agx::Matrix4x4 m2 = view.transform<agx::Matrix4x4 > ();
180 
181  if (m2 != m_glCamera->getViewMatrix())
182  m_glCamera->setViewMatrix(m2);
183 
184  auto viewport = m_mainCamera->getViewport();
185  m_glCamera->setViewPort(static_cast<int>(viewport->width()), static_cast<int>(m_mainCamera->getViewport()->height()));
186  }
187  using osg::Camera::DrawCallback::operator();
188

190  osg::Camera *m_mainCamera;
