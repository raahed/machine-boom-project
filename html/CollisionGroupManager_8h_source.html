282  void insert(agx::Physics::CollisionGroupSetPtr set)
283  {
284  agxAssert(set);
285  const GroupVector& groups = set.groups();
286 
287  agx::UInt32 hashValue = calculateHash(groups.begin(), groups.size());
288 
289  Bucket *bucket;
290  Bucket *targetBucket = 0;
291  size_t offset = 0;
292  // std::cout << "Insert " << cellIndex << " with id " << cellId << std::endl;
293 
294  /* Check if we should grow and rebuild hash table */
295  if (m_size >= (size_t)((float)m_capacity * AGX_HASH_SET_GROW_THRESHOLD))
296  this->rebuild((size_t)((float)(m_capacity+1) * AGX_HASH_SET_GROW_FACTOR));
297 
298  /* Iterate until target bucket is found */
299  bool goon = true;
300  do
301  {
302  bucket = &m_buckets[(hashValue + offset * offset) % m_capacity];
303 
304  /* Check if overwriting previous insertion with same key */
305  if (bucket->isValid() && compareGroups(groups.begin(), groups.size(), bucket->groupSet))
306  {
307  targetBucket = bucket;
308  break;
309  }
310 
311  /* Tag first free slot */
312  if (!bucket->isValid() && !targetBucket)
313  targetBucket = bucket;
314 
315  /* If we have found a free bucket we can stop iterating when we are sure that the key was not previously inserted */
316  if (targetBucket && offset >= m_maxProbeLength)
317  break;
318 
319  /* Go to next bucket */
320  offset++;
321 
322  /* Resize and rebuild table if it takes too long to find a free slot */
323  if (offset > AGX_HASH_SET_MAX_PROBE_LENGTH)
324  {
325  this->rebuild((size_t)((float)m_capacity * AGX_HASH_SET_GROW_FACTOR));
326  return this->insert(set);
327  }
328  } while (goon);
329 
330  agxAssert(targetBucket);
331 
332  if (offset > m_maxProbeLength)
333  m_maxProbeLength = (agx::UInt16)offset;
334 
335  targetBucket->groupSet = set;
336 
337  /* Check if inserting a new value rather than overwriting an old entry */
338  if (!targetBucket->isValid())
339  {
340  m_size++;
341  targetBucket->state = Bucket::VALID;
342  }
343  }

370  AGX_FORCE_INLINE agx::UInt32 findBucket(const agx::Physics::CollisionGroupPtr *localGroups, size_t numGroups) const
371  {
372  agxAssert(m_buckets);
373  if (m_buckets) {
374 
375  agx::UInt32 hashValue = calculateHash(localGroups, numGroups);
376 
377  size_t index = hashValue % m_capacity;
378  const Bucket *bucket = &m_buckets[index];
379 
380  /* First test is outside for-loop, hopefully getting better branch prediction */
381  if (bucket->isValid() && compareGroups(localGroups, numGroups, bucket->groupSet))
382  return (agx::UInt32)index;
383 
384 
385  for (size_t offset = 1; bucket->isActive() && offset <= m_maxProbeLength; offset++)
386  {
387  index = (hashValue + offset * offset) % m_capacity;
388  bucket = &m_buckets[index];
389 
390  if (bucket->isValid() && compareGroups(localGroups, numGroups, bucket->groupSet))
391  return (agx::UInt32)index;
392  }
393  }
394 
395  /* Not found */
396  return agx::InvalidIndex;
397  }

427  void rebuild(size_t size)
428  {
429  size = std::max( size, size_t(AGX_HASH_SET_MIN_SIZE) );
430  //LOGGER_DEBUG() << "HashTable: Rebuilding and resizing from " << numBuckets() << " to " << agx::nextPrime((int)(numBuckets() * resizeFactor)) << " buckets.\n" << LOGGER_END();
431  agxAssert(size >= m_size);
432  size_t oldNumBuckets = m_capacity;
433  Bucket *oldBuckets = m_buckets;
434 
435  m_capacity = agx::nextPrime((int)size);
436  m_buckets = (Bucket *)m_allocator.allocateBytes(m_capacity * sizeof(Bucket));
437  for (size_t i = 0; i < m_capacity; ++i)
438  ::new (&m_buckets[i]) Bucket();
439 
440  m_size = 0;
441  m_maxProbeLength = 0;
442 
443  for (size_t i = 0; i < oldNumBuckets; i++)
444  {
445  Bucket& bucket = oldBuckets[i];
446 
447  if (bucket.isValid())
448  this->insert(bucket.groupSet);
449  }
450 
451  m_allocator.deallocateBytes((void *)oldBuckets, oldNumBuckets * sizeof(Bucket));
452  }
453 
454  void deallocateBuckets()
455  {
456  m_allocator.deallocateBytes(m_buckets, m_capacity * sizeof(Bucket));
457  }
