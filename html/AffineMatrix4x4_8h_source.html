462  template <typename T>
463  AGX_FORCE_INLINE Vec3T<T> AffineMatrix4x4T<T>::getInvTranslate() const
464  {
465  T x = this->m_data[3][0];
466  T y = this->m_data[3][1];
467  T z = this->m_data[3][2];
468 
469  Vec3T<T> invTranslate;
470 
471  invTranslate[0] = -( x * this->m_data[0][0] + y* this->m_data[0][1] + z * this->m_data[0][2]);
472  invTranslate[1] = -( x * this->m_data[1][0] + y* this->m_data[1][1] + z * this->m_data[1][2]);
473  invTranslate[2] = -( x * this->m_data[2][0] + y* this->m_data[2][1] + z * this->m_data[2][2]);
474 
475  return invTranslate;
476  }

710  template <typename T>
711  AGX_FORCE_INLINE void AffineMatrix4x4T<T>::mult( const AffineMatrix4x4T<T>& lhs, const AffineMatrix4x4T<T>& rhs )
712  {
713  agx::prefetch<agx::L1>( lhs.m_data );
714  agx::prefetch<agx::L1>( rhs.m_data );
715 
716  if (&lhs==this)
717  {
718  postMult(rhs);
719  return;
720  }
721  if (&rhs==this)
722  {
723  preMult(lhs);
724  return;
725  }
726 
727  this->m_data[0][0] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3(lhs, rhs, 0, 0);
728  this->m_data[0][1] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3(lhs, rhs, 0, 1);
729  this->m_data[0][2] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3(lhs, rhs, 0, 2);
730 
731  this->m_data[1][0] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3(lhs, rhs, 1, 0);
732  this->m_data[1][1] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3(lhs, rhs, 1, 1);
733  this->m_data[1][2] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3(lhs, rhs, 1, 2);
734 
735  this->m_data[2][0] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3(lhs, rhs, 2, 0);
736  this->m_data[2][1] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3(lhs, rhs, 2, 1);
737  this->m_data[2][2] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3(lhs, rhs, 2, 2);
738 
739  this->m_data[3][0] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3(lhs, rhs, 3, 0) + rhs.m_data[3][0];
740  this->m_data[3][1] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3(lhs, rhs, 3, 1) + rhs.m_data[3][1];
741  this->m_data[3][2] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3(lhs, rhs, 3, 2) + rhs.m_data[3][2];
742 
743  }

745  template <typename T>
746  AGX_FORCE_INLINE void AffineMatrix4x4T<T>::preMult( const AffineMatrix4x4T<T>& other )
747  {
748 
749  // more efficient method just use a T[4] for temporary storage.
750  T t[4];
751  for(size_t col=0; col<3; ++col) {
752  t[0] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3( other, *this, 0, col );
753  t[1] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3( other, *this, 1, col );
754  t[2] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3( other, *this, 2, col );
755  t[3] = AGX_AFFINEMATRIX4X4_INNER_PRODUCT_3( other, *this, 3, col );
756  this->m_data[0][col] = t[0];
757  this->m_data[1][col] = t[1];
758  this->m_data[2][col] = t[2];
759  this->m_data[3][col] = t[3] + this->m_data[3][col];
760  }
761  }

763  template <typename T>
764  AGX_FORCE_INLINE void AffineMatrix4x4T<T>::postMult( const AffineMatrix4x4T<T>& other )
765  {
766 
767  // more efficient method just use a T[3] for temporary storage.
768  T t[3];
769  for(size_t row=0; row<3; ++row)
770  {

877  template <typename T>
878  AGX_FORCE_INLINE bool AffineMatrix4x4T<T>::invert( const AffineMatrix4x4T<T>& rhs)
879  {
880  T x,y,z;
881 
882  // diagonal
883  this->m_data[0][0] = rhs.m_data[0][0];
884  this->m_data[1][1] = rhs.m_data[1][1];
885  this->m_data[2][2] = rhs.m_data[2][2];
886 
887  // rotation
888  this->m_data[0][1] = rhs.m_data[1][0];
889  this->m_data[1][0] = rhs.m_data[0][1];
890  this->m_data[0][2] = rhs.m_data[2][0];
891  this->m_data[2][0] = rhs.m_data[0][2];
892  this->m_data[1][2] = rhs.m_data[2][1];
893  this->m_data[2][1] = rhs.m_data[1][2];
894 
895  x = rhs.m_data[3][0];
896  y = rhs.m_data[3][1];
897  z = rhs.m_data[3][2];
898 
899  // translation
900  this->m_data[3][0] = -( x * this->m_data[0][0] + y* this->m_data[1][0] + z * this->m_data[2][0]);
901  this->m_data[3][1] = -( x * this->m_data[0][1] + y* this->m_data[1][1] + z * this->m_data[2][1]);
902  this->m_data[3][2] = -( x * this->m_data[0][2] + y* this->m_data[1][2] + z * this->m_data[2][2]);
903 
904  return true;
905  }
