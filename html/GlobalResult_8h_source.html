339  AGX_FORCE_INLINE bool GlobalResult::Transaction::commit(Index sortIndex)
340  {
341  m_target->getMutex().lock();
342  auto currentActiveSet = m_target->getActiveSet();
343  m_target->getMutex().unlock();
344 
345  if (m_bufferSet == currentActiveSet)
346  {
347  #ifdef AGX_DEBUG
348  m_target->reportCompletedTransaction(this);
349  #endif
350 
351  agxAssert(sortIndex < m_target->m_sortedRanges.size());
352  m_target->m_sortedRanges[sortIndex] = m_range;
353 
354  return true;
355  }
356 
357  m_bufferSet = m_target->reportFailedTransaction(this);
358  return false;
359  }
360 
361  AGX_FORCE_INLINE bool GlobalResult::Transaction::commit()
362  {
363  agxAssertN(!(m_target->m_settings & (SORT|ORDER)) || m_target->m_sortedRanges.empty(), "%s: Must use extended commit method if SORT or ORDER is specified", m_target->getPath().c_str());
364 
365  m_target->getMutex().lock();
366  BufferSet *currentActiveSet = m_target->getActiveSet();
367  m_target->getMutex().unlock();
368 
369  if (m_bufferSet == currentActiveSet)
370  {
371  #ifdef AGX_DEBUG
372  m_target->reportCompletedTransaction(this);
373  #endif
374 
375  return true;
376  }
377 
378  m_bufferSet = m_target->reportFailedTransaction(this);
379  return false;
