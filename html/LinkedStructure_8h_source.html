568  template<typename T>
569  T* LinkedStructure::getSegment( agx::Real restLengthFromStart ) const
570  {
571  if ( m_segments.empty() )
572  return nullptr;
573 
574  if ( restLengthFromStart <= 0.0 )
575  return m_segments.front()->template as<T>();
576 
577  agx::Real restLengthAlongStructure = 0.0;
578  for ( auto segment : this->template getSegments<T>() ) {
579  if ( restLengthAlongStructure + segment->getLength() < restLengthFromStart )
580  restLengthAlongStructure += segment->getLength();
581  else
582  return segment->template as<T>();
583  }
584 
585  return m_segments.back()->template as<T>();
586  }
