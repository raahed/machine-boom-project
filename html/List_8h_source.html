803  {

907  template< typename Pred >
908  void sort( Pred pred )
909  {
910  if ( m_size < 2 )
911  return;
912 
913  const size_type MAXBINS = 25;
914  my_type tmpList;
915  my_type binList[ MAXBINS + 1 ];
916 
917  size_type maxBin = 0;
918  while ( !empty() ) {
919  tmpList.splice( tmpList.begin(), *this, begin(), ++begin(), 1 );
920 
921  size_type bin = 0;
922  for ( ; bin < maxBin && !binList[ bin ].empty(); ++bin ) {
923  binList[ bin ].merge( tmpList, pred );
924  binList[ bin ].swap( tmpList );
925  }
926 
927  if ( bin == MAXBINS )
928  binList[ bin - 1 ].merge( tmpList, pred );
929  else {
930  binList[ bin ].swap( tmpList );
931  maxBin += size_type( bin == maxBin );
932  }
933  }
934 
935  for ( size_type bin = 1; bin < maxBin; ++bin )
936  binList[ bin ].merge( binList[ bin - 1 ], pred );
937 
938  splice( end(), binList[ maxBin - 1 ], binList[ maxBin - 1 ].begin(), binList[ maxBin - 1 ].end(), binList[ maxBin - 1 ].size() );
939  }

949  void splice( const_iterator _where, my_type& other, const_iterator first, const_iterator last, size_type count )
950  {
951  if ( count == 0 )
952  return;
953 
954  if ( this != &other ) {
955  m_size += count;
956  other.m_size -= count;
957  }
958 
959  // If first is other.begin(), other gets a new m_front.
960  if ( first.m_ptr->m_prev == nullptr )
961  other.m_front = last.m_ptr;
962  else
963  first.m_ptr->m_prev->m_next = last.m_ptr;
964 
965  last.m_ptr->m_prev->m_next = _where.m_ptr;
966 
967  if ( _where.m_ptr->m_prev == nullptr )
968  m_front = first.m_ptr;
969  else
970  _where.m_ptr->m_prev->m_next = first.m_ptr;
971
