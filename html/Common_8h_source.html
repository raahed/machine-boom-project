216  AGX_FORCE_INLINE UInt8 calculateTier( Real minCellSize, const agxData::Array<Real>& gridTier_size )
217  {
218  size_t numTiers = gridTier_size.size();
219  if ( numTiers == 0 )
220  return InvalidIndex;
221 
222  UInt8 tier;
223  #if 0
224  // Experimental log2 way of finding the correct tier. Might be slower than
225  // the short loop below and I'm worried about precision problems. Also, if
226  // the required minimum cell size is larger than the currently largest
227  // tier, then this code will (probably) return the smallest (numTiers-1)
228  // tier.
229  // Don't think we should use this. Comments?
230  tier = (UInt8)( log2(largestCellSize) - log2(minCellSize) );
231  if ( tier >= numTiers )
232  tier = numTiers-1;
233  #else
234  // Safe loop for finding the correct tier. Searches down the hierarchy.
235  // May produce the largest tier even if even it is too small. Also,
236  // will produce the smallest tier even if the object really should be
237  // in a even smaller tier.
238  for ( tier = 0; tier < numTiers-1; ++tier )
239  {
240  // Test if the NEXT tier would be too small.
241  if ( gridTier_size[ tier+1 ] < minCellSize )
242  break;
243  }
244  #endif
245 
246  // Make sure we found a tier that isn't too small.
247  if ( minCellSize > gridTier_size[ tier ] )
248  tier = InvalidIndex;
249 
250  // Make sure the object shouldn't fit better in a even smaller tier.
251  // NOTE: Important that there is only one valid tier for each requested cell size.
252  // We must therefore use '<=' instead of '<' to prevent ambiguous tier assignments.
253  else if ( Real(2) * minCellSize <= gridTier_size[ tier ] )
254  tier = InvalidIndex;
255 
256  return tier;
257  }
