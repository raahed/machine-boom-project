40  // From chapter 7.3.5 of Christer Ericson's book "Real time collision detection"
41  // Is approximately 3x faster than naive loop implementation.
42  AGX_FORCE_INLINE UInt32 morton_part1By2(UInt32 n)
43  {
44  n = n & 0x3FF; // Only keep low 10 bits
45  n = (n | (n << 16)) & 0xFF0000FF;
46  n = (n | (n << 8)) & 0x0300F00F;
47  n = (n | (n << 4)) & 0x030C30C3;
48  n = (n | (n << 2)) & 0x09249249;
49  return n;
50  }

53  AGX_FORCE_INLINE UInt32 morton(UInt32 x, UInt32 y, UInt32 z)
54  {

64  // Reference implementation
65  AGX_FORCE_INLINE UInt32 morton_reference(UInt32 x, UInt32 y, UInt32 z)
66  {
67  UInt32 result = 0;
68 
69  for(size_t i = 0; i < 10; i++)
70  {
71  UInt32 mask = 1 << i;
72  result |= (x & mask) << (i * 2) | (y & mask) << (i * 2 + 1) | (z & mask) << (i * 2 + 2);
73  }
74 
75  return result;
76  }
