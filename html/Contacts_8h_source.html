553  AGX_FORCE_INLINE agx::Real ContactPoint::getForceMagnitude() const
554  {
555  agx::Real sqr =
556  localForce()[ NORMAL_FORCE ] * localForce()[ NORMAL_FORCE ] +
557  localForce()[ TANGENTIAL_FORCE_U ] * localForce()[ TANGENTIAL_FORCE_U ] +
558  localForce()[ TANGENTIAL_FORCE_V ] * localForce()[ TANGENTIAL_FORCE_V ];
559  return sqr > 0 ? std::sqrt( sqr ) : 0;
560  }

668  AGX_FORCE_INLINE agx::Vec3f GeometryContact::calculateRelativeVelocity(size_t pointIndex) const
669  {
670  agx::Vec3 v1, v2;
671  if (pointIndex > points().size() - 1)
672  return agx::Vec3f();
673 
674  const agx::RigidBody* rb1 = geometry(0)->getRigidBody();
675  const agx::RigidBody* rb2 = geometry(1)->getRigidBody();
676 
677  if (rb1)
678  v1 = rb1->getVelocity() + (rb1->getAngularVelocity() ^ (points()[pointIndex].point() - rb1->getCmPosition()));
679  if (rb2)
680  v2 = rb2->getVelocity() + (rb2->getAngularVelocity() ^ (points()[pointIndex].point() - rb2->getCmPosition()));
681 
682  return static_cast<agx::Vec3f>(v1 - v2);
683  }
