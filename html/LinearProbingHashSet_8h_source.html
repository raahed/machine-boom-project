436  // Remove all elements, keeping full capacity.
437  void clear()
438  {
439  for (size_t bucket=0; bucket<_num_buckets; ++bucket) {
440  if (_states[bucket] == State::FILLED) {

449  // Make room for this many elements
450  void reserve(size_t num_elems)
451  {
452  size_t required_buckets = num_elems + num_elems/2 + 1;
453  if (required_buckets <= _num_buckets) {
454  return;
455  }
456  size_t num_buckets = 4;
457  while (num_buckets < required_buckets) { num_buckets *= 2; }
458 
459  auto new_states = (State*)malloc(num_buckets * sizeof(State));
460  auto new_keys = (KeyT*)malloc(num_buckets * sizeof(KeyT));
461 
462  if (!new_states || !new_keys) {
463  free(new_states);
464  free(new_keys);
465  throw std::bad_alloc();
466  }
467 
468  // auto old_num_filled = _num_filled;
469  auto old_num_buckets = _num_buckets;
470  auto old_states = _states;
471  auto old_keys = _keys;
472 
473  _num_filled = 0;
474  _num_buckets = num_buckets;
475  _mask = _num_buckets - 1;
476  _states = new_states;
477  _keys = new_keys;
478 
479  std::fill_n(_states, num_buckets, State::INACTIVE);
480 
481  _max_probe_length = -1;
482 
483  for (size_t src_bucket=0; src_bucket<old_num_buckets; src_bucket++) {
484  if (old_states[src_bucket] == State::FILLED) {
485  auto& src = old_keys[src_bucket];
486 
487  auto dst_bucket = find_empty_bucket(src);
488  agxAssert(dst_bucket != (size_t)-1);
489  agxAssert(_states[dst_bucket] != State::FILLED);
490  _states[dst_bucket] = State::FILLED;
491  new(_keys + dst_bucket) KeyT(std::move(src));
492  _num_filled += 1;
493 
494  src.~KeyT();
495  }
496  }
497 
498  // DCHECK_EQ_F(old_num_filled, _num_filled);
499 
500  free(old_states);
501  free(old_keys);
502  }

504  protected:
505  // Find the bucket with this key, or return nullptr
506  template<typename T2>
507  size_t find_filled_bucket(const T2& key) const
508  {
509  if (empty()) { return (size_t)-1; } // Optimization
510 
511  auto hash_value = _hasher(key);
512  for (int offset=0; offset<=_max_probe_length; ++offset) {
513  auto bucket = (hash_value + offset) & _mask;
514  if (_states[bucket] == State::FILLED && agx::hashKeyEqual(_keys[bucket], key)) {
515  return bucket;
516  }
517  if (_states[bucket] == State::INACTIVE) {
518  return (size_t)-1; // End of the chain!
519  }
520  }
521  return (size_t)-1;
522  }

524  void erase_bucket(size_t bucket) {
525  _states[bucket] = State::ACTIVE;
526  _keys[bucket].~KeyT();
527  _num_filled -= 1;
528  }

537  // Find the bucket with this key, or return a good empty bucket to place the key in.
538  // In the latter case, the bucket is expected to be filled.
539  size_t find_or_allocate(const KeyT& key)
540  {
541  auto hash_value = _hasher(key);
542  size_t hole = (size_t)-1;
543  int offset=0;
544  for (; offset<=_max_probe_length; ++offset) {
545  auto bucket = (hash_value + offset) & _mask;
546 
547  if (_states[bucket] == State::FILLED) {
548  if (agx::hashKeyEqual(_keys[bucket], key)) {
549  return bucket;
550  }
551  } else if (_states[bucket] == State::INACTIVE) {
552  return bucket;
553  } else {
554  // ACTIVE: keep searching
555  if (hole == (size_t)-1) {
556  hole = bucket;
557  }
558  }
559  }
560 
561  // No key found - but maybe a hole for it
562 
563  agxAssert(offset == _max_probe_length+1);
564 
565  if (hole != (size_t)-1) {
566  return hole;
567  }
568 
569  // No hole found within _max_probe_length
570  for (; ; ++offset) {
571  auto bucket = (hash_value + offset) & _mask;
572 
573  if (_states[bucket] != State::FILLED) {
574  _max_probe_length = offset;
575  return bucket;
576  }
577  }
578  }
579 
580  // key is not in this map. Find a place to put it.
581  size_t find_empty_bucket(const KeyT& key)
582  {
583  auto hash_value = _hasher(key);
584  for (int offset=0; ; ++offset) {
585  auto bucket = (hash_value + offset) & _mask;
586  if (_states[bucket] != State::FILLED) {
587  if (offset > _max_probe_length) {
588  _max_probe_length = offset;
589  }
590  return bucket;
591  }
592  }
593  }

597  State* _states = nullptr;
598  KeyT* _keys = nullptr;
599  size_t _num_buckets = 0;
600  size_t _num_filled = 0;
601  int _max_probe_length = -1; // Our longest bucket-brigade is this long. ONLY when we have zero elements is this ever negative (-1).
602  size_t _mask = 0; // _num_buckets minus one
