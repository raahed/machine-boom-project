521  insert_iterator insert(const KeyT& key, const ValueT value)
522  {
523  check_expand_need();
524 
525  auto bucket = find_or_allocate(key);
526  bool didOverwrite;
527  if (_states[bucket] == State::FILLED) {
528  (_pairs + bucket)->second = value;
529  didOverwrite = true;
530  } else {
531  _states[bucket] = State::FILLED;
532  new(_pairs + bucket) PairT(key, value);
533  _num_filled++;
534  didOverwrite = false;
535  }
536 
537  return insert_iterator(this, bucket, didOverwrite);
538  }

656  void clear(int /*IGNORED_AGX_PARAMETER*/ = 0)
657  {
658  for (size_t bucket=0; bucket<_num_buckets; ++bucket) {

671  void reserve(size_t num_elems)
672  {
673  size_t required_buckets = num_elems + num_elems/2 + 1;
674  if (required_buckets <= _num_buckets) {
675  return;
676  }
677  size_t num_buckets = 4;
678  while (num_buckets < required_buckets) { num_buckets *= 2; }
679 
680  auto new_states = (State*)_allocator.allocateBytes(num_buckets * sizeof(State));
681  auto new_pairs = (PairT*)_allocator.allocateBytes(num_buckets * sizeof(PairT));
682 
683  if (!new_states || !new_pairs) {
684  _allocator.deallocateBytes(new_states, num_buckets * sizeof(State));
685  _allocator.deallocateBytes(new_pairs, num_buckets * sizeof(PairT));
686  throw std::bad_alloc();
687  }
688 
689  //auto old_num_filled = _num_filled;
690  auto old_num_buckets = _num_buckets;
691  auto old_states = _states;
692  auto old_pairs = _pairs;
693 
694  _num_filled = 0;
695  _num_buckets = num_buckets;
696  _mask = _num_buckets - 1;
697  _states = new_states;
698  _pairs = new_pairs;
699 
700  std::fill_n(_states, num_buckets, State::INACTIVE);
701 
702  _max_probe_length = -1;
703 
704  for (size_t src_bucket=0; src_bucket<old_num_buckets; src_bucket++) {
705  if (old_states[src_bucket] == State::FILLED) {
706  auto& src_pair = old_pairs[src_bucket];
707 
708  auto dst_bucket = find_empty_bucket(src_pair.first);
709  agxAssert(dst_bucket != (size_t)-1);
710  agxAssert(_states[dst_bucket] != State::FILLED);
711  _states[dst_bucket] = State::FILLED;
712  new(_pairs + dst_bucket) PairT(std::move(src_pair));
713  _num_filled += 1;
714 
715  src_pair.~PairT();
716  }
717  }
718 
719  //agxAssert(old_num_filles == _num_filled);
720 
721  _allocator.deallocateBytes(old_states, old_num_buckets * sizeof(State));
722  _allocator.deallocateBytes(old_pairs, old_num_buckets * sizeof(PairT));
723  }

734 protected:
735  // Find the bucket with this key, or return nullptr
736  template<typename T2>
737  size_t find_filled_bucket(const T2& key) const
738  {
739  if (empty()) { return (size_t)-1; } // Optimization
740 
741  auto hash_value = _hasher(key);
742  for (int offset=0; offset<=_max_probe_length; ++offset) {
743  auto bucket = (hash_value + offset) & _mask;
744  if (_states[bucket] == State::FILLED && agx::hashKeyEqual(_pairs[bucket].first, key)) {
745  return bucket;
746  }
747  if (_states[bucket] == State::INACTIVE) {
748  return (size_t)-1; // End of the chain!
749  }
750  }
751  return (size_t)-1;
752  }
753 
754  void erase_bucket(size_t bucket) {
755  _states[bucket] = State::ACTIVE;
756  _pairs[bucket].~PairT();
757  _num_filled -= 1;
758  }

761 private:
762  // Can we fit another element?
763  void check_expand_need()
764  {
765  reserve(_num_filled + 1);
766  }
767  // Find the bucket with this key, or return a good empty bucket to place the key in.
768  // In the latter case, the bucket is expected to be filled.
769  size_t find_or_allocate(const KeyT& key)
770  {
771  auto hash_value = _hasher(key);
772  size_t hole = (size_t)-1;
773  int offset=0;
774  for (; offset<=_max_probe_length; ++offset) {
775  auto bucket = (hash_value + offset) & _mask;
776 
777  if (_states[bucket] == State::FILLED) {
778  if (agx::hashKeyEqual(_pairs[bucket].first, key)) {
779  return bucket;
780  }
781  } else if (_states[bucket] == State::INACTIVE) {
782  return bucket;
783  } else {
784  // ACTIVE: keep searching
785  if (hole == (size_t)-1) {
786  hole = bucket;
787  }
788  }
789  }
790 
791  // No key found - but maybe a hole for it
792 
793  agxAssert(offset == _max_probe_length+1);
794 
795  if (hole != (size_t)-1) {
796  return hole;
797  }
798 
799  // No hole found within _max_probe_length
800  for (; ; ++offset) {
801  auto bucket = (hash_value + offset) & _mask;
802 
803  if (_states[bucket] != State::FILLED) {
804  _max_probe_length = offset;
805  return bucket;
806  }
807  }
808  }
809 
810  // key is not in this map. Find a place to put it.
811  size_t find_empty_bucket(const KeyT& key)
812  {
813  auto hash_value = _hasher(key);
814  for (int offset=0; ; ++offset) {
815  auto bucket = (hash_value + offset) & _mask;
816  if (_states[bucket] != State::FILLED) {
817  if (offset > _max_probe_length) {
818  _max_probe_length = offset;
819  }
820  return bucket;
821  }
822  }
823  }
824 
825 private:
826  HashT _hasher;
827  AllocatorT _allocator = AllocatorT();
828  State* _states = nullptr;
829  PairT* _pairs = nullptr;
830  size_t _num_buckets = 0;
831  size_t _num_filled = 0;
832  int _max_probe_length = -1; // Our longest bucket-brigade is this long. ONLY when we have zero elements is this ever negative (-1).
833  size_t _mask = 0; // _num_buckets minus one
834 };
