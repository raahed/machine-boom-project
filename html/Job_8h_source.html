517  AGX_FORCE_INLINE void Job::removeDependency(Job *parent)
518  {
519  agxAssert(parent);
520  // agxAssert(parent->m_children.contains(this));
521  if (parent->m_children.findAndErase(this)) {
522  m_numDependencies--;
523  #ifdef AGX_DEBUG
524  agxVerify(m_parents.findAndErase(parent));
525  #endif
526  }
