218  AGX_FORCE_INLINE Index calculateNeighborIndex(const Vec3i& offset)
219  {
220  agxAssert(isNeighborOffset(offset));
221  Vec3i shiftedOffset = offset + 1;
222  Index neighborIndex = (Index)((shiftedOffset[0] * 9) + (shiftedOffset[1] * 3) + shiftedOffset[2]);
223  agxAssert(getNeighborOffset(neighborIndex) == offset);
224  return neighborIndex;
225  }

397  AGX_FORCE_INLINE UInt8 calculateZoneType(Vec3i id, UInt tierAccumulation)
398  {
399  Int mask = (Int)generateBitMask(tierAccumulation);
400 
401  id[0] &= mask;
402  id[1] &= mask;
403  id[2] &= mask;
404 
405  UInt zoneType = 0;
406  zoneType += (id[0] == mask ? 18 : (id[0] == 0 ? 0 : 9));
407  zoneType += (id[1] == mask ? 6 : (id[1] == 0 ? 0 : 3));
408  zoneType += (id[2] == mask ? 2 : (id[2] == 0 ? 0 : 1));
409 
410  agxAssert(zoneType < 27);
411  return (UInt8)zoneType;
412  }
