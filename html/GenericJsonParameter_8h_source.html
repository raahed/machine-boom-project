133  void parseVec3( const agxJson::Value& value, agx::Vec3& result ) const
134  {
135 
136  if ( value.isArray() ) {
137  if ( value.size() != m_numValues )
138  LOGGER_WARNING() << "Array with values for " << m_name << " does not have expected length "
139  << m_numValues << "\n" << LOGGER_END();
140 
141  agxJson::Value::ArrayIndex maxIndex = std::min(value.size(), std::min(m_numValues, 3u));
142 
143  for ( agxJson::Value::ArrayIndex i = 0; i < maxIndex; i++) {
144  agxJson::Value valueNode = value[i];
145 
146  if ( !isDefault(valueNode) ) {
147  result[i] = parseReal( valueNode );
148  }
149  }
150 
151  }
152  else if ( value.isDouble() ) {
153  // If the value is a single number, then we return a Vec3 with said value
154  result = agx::Vec3( value.asDouble() );
155  }
156  else if ( value.isString() ) {
157  if ( !isDefault(value) )
158  result = agx::Vec3( parseReal( value ) );
159  }
160  else {
161  LOGGER_WARNING() << "Unknown value \"" << value.asString() << "\" in json data for " <<
162  getParameterPath() << LOGGER_ENDL();
163  }
164  }

171  bool readProperty(const agxJson::Value& root, T* material )
172  {
173  agxJson::Value object;
174 
175  if ( m_group != "" ) {
176  if ( !root.hasMember( m_group.c_str() ) ) {
177  LOGGER_WARNING() << "Json file is missing object " << m_group << LOGGER_ENDL();
178  return false;
179  }
180  object = root[m_group];
181  }
182  else {
183  object = root;
184  }
185 
186  if ( !object.hasMember( m_name.c_str() ) ) {
187  LOGGER_WARNING() << "Json object is missing " << m_name << LOGGER_ENDL();
188  return false;
189  }
190 
191  agxJson::Value valueNode = object[m_name];
192 
193  // Resolved, the item is found as "valueNode" in the json data.
194  // There can still be warnings from the parsing
195  m_resolved = true;
196 
197  if ( m_type == ARRAY_VEC3 ) {
198  // Need to read previous values in case of partial use of default,
199  // e.g. [1, "default", 2]
200  agx::Vec3 v = m_getVec3( material );
201 
202  parseVec3( object[m_name], v );
203 
204  m_setVec3( v, material );
205  }
206  else if ( m_type == SCALAR_REAL) {
207  if ( !isDefault(valueNode)) {
208  agx::Real r = parseReal( valueNode );
209  m_setReal( r, material );
210  }
211  }
212  else if ( m_type == SCALAR_BOOL ) {
213  if ( !isDefault(valueNode)) {
214  bool b = parseBool( valueNode );
215  m_setBool( b, material );
216  }
217  }
218  else {
219  LOGGER_WARNING() << "Unknown parameter type for " << getParameterPath() << LOGGER_ENDL();
220  return false;
221  }
222 
223  return true;
224  }

270  bool writeProperty(agxJson::Value& root, const T* material )
271  {
272  agxJson::Value* outputNode = &root;
273 
274  if ( m_group.length() > 0 ) {
275 
276  if ( !root.hasMember( m_group ) ) {
277  return false;
278  }
279 
280  outputNode = &root[ m_group ];
281  }
282 
283 
284  if (m_type == SCALAR_REAL ) {
285  writeValueReal( (*outputNode)[ m_name ], m_getReal( material ) );
286  }
287  else if ( m_type == SCALAR_BOOL ) {
288  writeValueBool( (*outputNode)[ m_name ], m_getBool( material ) );
289  }
290  else if ( m_type == ARRAY_VEC3 ) {
291  writeValueVec3( (*outputNode)[ m_name ], m_getVec3( material ) );
292  }
293  else {
294  LOGGER_WARNING() << "Unknown parameter type for " << getParameterPath() << LOGGER_ENDL();
295  }
296 
297  m_resolved = outputNode->hasMember( m_name );
298 
299  return m_resolved;
300  }
