178  void readRef(T** obj, StorageMode mode)
179  {
180  if (mode == STORE_BY_REF) {
181 
182  Serializable* sObj;
183 
184  sObj = readObjectByRef();
185 
186  *obj = dynamic_cast<T*>(sObj);
187  if (sObj != 0 && *obj == 0) {
188  agx::String str = agx::String::format("Type mismatch: read type \'%s\' Expected \'%s\'",
189  sObj->getClassName(),
190  T::getConstructClassId());
191 
192  agxThrow ArchiveException(str);
193  }
194  }
195  else
196  {
197  agxThrow ArchiveException("Invalid storage mode specified for read");
198  }
199  }
