301  template<typename T>
302  void OrientedFrictionModelImpl<T>::calculateTangentPlane( const agxCollide::Geometry* /*geometry1*/,
303  const agxCollide::Geometry* /*geometry2*/,
304  const agx::Vec3& /*point*/,
305  const agx::Vec3& normal,
306  const agx::Real /*depth*/,
307  agx::Vec3& ret_u,
308  agx::Vec3& ret_v ) const
309  {
310  // If the frame is null we treat the primary direction as given in world frame.
311  ret_u = m_refFrame != nullptr ?
312  m_refFrame->transformVectorToWorld( m_primaryDirection ) :
313  m_primaryDirection;
314  ret_u -= ( normal * ret_u ) * normal;
315  if ( ret_u.normalize() < agx::RealEpsilon ) {
316  agx::Attachment::createAttachmentBase( normal, ret_u, ret_v );
317  return;
318  }
319 
320  ret_v = normal.cross( ret_u );
321  }
