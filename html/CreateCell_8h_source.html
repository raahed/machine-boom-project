87  template <typename GridTierDataT, typename CellDataT, typename CellT, typename CellIdT, typename CellPosT, typename GridCellTableT, size_t NumNeighbors>
88  UInt32 createCell
89  (
90  agxData::Array< agx::UInt32 >& deadCells,
91  agxData::Array< Physics::HierarchicalGrid::ContactZonePtr >& emptyZones,
92  UInt8 tier,
93  const CellIdT cellId,
94  GridTierDataT& gridTier,
95  CellDataT& cell,
96  agxData::Array< CellT >& rootCells,
97  const CellIdT *NeighborTable,
98  agx::Bool& hasNewTiers,
99  Physics::HierarchicalGrid::SolveBodyManager* solveBodyManager,
100  const agx::UInt& contactZoneAccumulationLevel
101  )
102  {
103  UInt32 parentCellIndex = InvalidIndex;
104 
105  if (tier > 0)
106  {
107  // Make sure parent cell is available, we want big tier cells first in list of new cells for better performance in InitializedNewCells kernel
108  UInt8 parentTier = (UInt8)(tier - 1);
109  CellIdT parentId = calculateParentCellId(cellId);
110 
111  GridCellTableT& parentTierTable = *gridTier.cellTable[ parentTier ];
112  parentCellIndex = parentTierTable.find( parentId );
113 
114  // Create parent cell if needed (recursive)
115  if ( parentCellIndex == InvalidIndex )
116  {
117  size_t oldNumTiers = gridTier.cellTable.size();
118 
119  parentCellIndex = createCell
120  <
121  GridTierDataT, CellDataT, CellT, CellIdT, CellPosT, GridCellTableT, NumNeighbors
122  >
123  (
124  deadCells, emptyZones, parentTier, parentId, gridTier, cell,
125  rootCells, NeighborTable, hasNewTiers, solveBodyManager,
126  contactZoneAccumulationLevel
127  );
128 
129  // Important: update current tier
130  tier = agx::UInt8(tier + (gridTier.numElements() - oldNumTiers));
131  }
132  }
133 
134  UInt32 cellIndex = InvalidIndex;
135 
136 
137  // Check if we have a dead cell to use
138  if (!deadCells.empty())
139  {
140  cellIndex = deadCells.back();
141  deadCells.range().end()--;
142  // std::cout << "Reusing dead cell " << cellIndex << " at index " << deadCells.size() << std::endl;
143 
144  // Must explicitly initialize parameters when reusing an old cell
145  cell.parent[cellIndex] = InvalidIndex;
146  cell.numChildren[cellIndex] = 0;
147  cell.invDepth[cellIndex] = 0;
148  }
149  else
150  {
151  cellIndex = cell.createInstance();
152  }
153 
154 
155 
156  cell.tier[ cellIndex ] = tier;
157  cell.id[ cellIndex ] = cellId;
158  cell.state[ cellIndex ] = CELL_UNINITIALIZED;
159 
160  GridCellTableT& cellTable = *gridTier.cellTable[ tier ];
161  agxAssert( cellTable.find( cellId ) == InvalidIndex );
162  cellTable.insert(cellId, cellIndex);
163 
164  // std::cout << cellIndex << ": Creating cell " << cellId << " in tier " << (UInt)tier << " with parent " << parentCellIndex << std::endl;
165 
166  if ( tier > 0 )
167  {
168  connectToParentCell<CellDataT, CellIdT>(cellIndex, cellId, parentCellIndex, cell, solveBodyManager, contactZoneAccumulationLevel);
169  }
170  else // Special handling of root cell
171  {
178  UInt rootIndex = calculateRootCellIndex(cellId);
179  // std::cout << "Creating root " << cellId << ", cellIndex: " << cellIndex << ", cell size: " << gridTier.size[tier] << std::endl;
180  solveBodyManager->registerCell(cell.instance[cellIndex]);
181 
182  if (!rootCells[rootIndex])
183  {
184  agx::Physics::HierarchicalGrid::CellPtr root = agxData::EntityPtr::createWithIndex(cell.getStorage(), cellIndex);
185  rootCells[rootIndex] = root;
186 
187  // Set up neighbor connections to other root cells, including self
188  UInt32 *neighborList = &cell.neighbors[CellDataT::neighborsArraySize * cellIndex];
189 
190  for (size_t i = 0; i < NumNeighbors; ++i)
191  neighborList[i] = InvalidIndex;
192 
193  for (size_t i = 0; i < rootCells.size(); ++i)
194  {
195  if (!rootCells[i])
196  continue;
197 
198  UInt32 otherIndex = rootCells[i].calculateIndex();
199 
200  CellIdT offset = cell.id[otherIndex] - cellId;
201  if (agx::Physics::HierarchicalGrid::isNeighborOffset(offset))
202  {
203  UInt localNeighborIndex = agx::Physics::HierarchicalGrid::calculateNeighborIndex(offset);
204  neighborList[localNeighborIndex] = otherIndex;
205 
206  UInt32 *reverseList = &cell.neighbors[CellDataT::neighborsArraySize * otherIndex];
207  reverseList[ NumNeighbors-1-localNeighborIndex ] = cellIndex;
208  }
209  }
210  }
211  else
212  {
213  // Remove current root cells
214  for (size_t i = 0; i < rootCells.size(); ++i)
215  rootCells[i] = agxData::EntityPtr();
216 
217  // Create new top tier
218  createMissingTiers
219  <
220  GridTierDataT, CellDataT, CellT, CellIdT, CellPosT, GridCellTableT, NumNeighbors
221  >
222  (
223  gridTier.size[0] * Real(2),
224  deadCells,
225  emptyZones,
226  gridTier,
227  cell,
228  rootCells,
229  NeighborTable,
230  hasNewTiers,
231  solveBodyManager,
232  contactZoneAccumulationLevel
233  );
234  }
235 
236  }
237 
238  return cellIndex;
239  }

242  template <typename CellDataT, typename CellIdT>
243  AGX_FORCE_INLINE void connectToParentCell
244  (
245  Index cellIndex,
246  const CellIdT cellId,
247  Index parentCellIndex,
248  CellDataT& cell,
249  Physics::HierarchicalGrid::SolveBodyManager* solveBodyManager,
250  const agx::UInt& contactZoneAccumulationLevel
251  )
252  {
253  // Connect both ways
254  // std::cout << "Connecting " << cellIndex << " with parent " << parentCellIndex << std::endl;
255  agxAssert(parentCellIndex != InvalidIndex);
256  cell.parent[ cellIndex ] = (UInt32)parentCellIndex;
257  UInt32 *childList = &cell.children[CellDataT::childrenArraySize * parentCellIndex];
258  UInt childIndex = calculateCellChildIndex(cellId);
259 
260  const UInt numChildren = cell.numChildren[ parentCellIndex ];
261  UInt compactedIndex = 0;
262  for (; compactedIndex < numChildren; ++compactedIndex)
263  {
264  if (childIndex < childList[ compactedIndex ])
265  break;
266  }
267 
268  for (UInt n = numChildren; n > compactedIndex; --n)
269  childList[n] = childList[n-1];
270 
271  childList[compactedIndex] = (UInt32)cellIndex;
272 
273  cell.numChildren[ parentCellIndex ]++;
274  agxAssert(cell.numChildren[parentCellIndex] <= CellDataT::childrenArraySize);
275 
276  // Update inv depth of parents if parent was a leaf node
277  if (numChildren == 0)
278  {
279  UInt8 invDepth = cell.invDepth[cellIndex];
280 
281  while (parentCellIndex != InvalidIndex && cell.invDepth[parentCellIndex] <= invDepth)
282  {
283  // std::cout << "cell " << cell.indexToId(parentCellIndex) << ", id: " << cell.id[parentCellIndex] << ", tier: " << (UInt)cell.tier[parentCellIndex] << " now has inv depth " << (UInt)(invDepth + 1) << std::endl;
284  cell.invDepth[parentCellIndex] = ++invDepth;
285 
286  if ( (invDepth >= contactZoneAccumulationLevel || cell.tier[parentCellIndex] <= contactZoneAccumulationLevel) && !cell.solveBodyOffsets[parentCellIndex])
287  solveBodyManager->registerCell(cell.instance[parentCellIndex]);
288 
289  parentCellIndex = cell.parent[parentCellIndex];
290  }
291  }
292  }
