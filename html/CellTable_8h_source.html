211  void insert(const CellIdT& cellId, UInt32 cellIndex)
212  {
213  UInt32 hashValue = m_hashFunctor(cellId);
214  Bucket *bucket;
215  Bucket *targetBucket = 0;
216  size_t offset = 0;
217  // std::cout << "Insert " << cellIndex << " with id " << cellId << std::endl;
218 
219  /* Check if we should grow and rebuild hash table */
220  if (m_size >= (size_t)((float)m_capacity * AGX_HASH_TABLE_GROW_THRESHOLD))
221  this->rebuild((size_t)((float)(m_capacity+1) * AGX_HASH_TABLE_GROW_FACTOR));
222 
223 #ifdef _MSC_VER
224 #pragma warning(push)
225 #pragma warning(disable: 4127) // warning C4127: conditional expression is constant
226 #endif
227 
228  {
229  /* Iterate until target bucket is found */
230  const bool trueVal = true;
231  do
232  {
233  bucket = &m_buckets[(hashValue + offset * offset) % m_capacity];
234 
235  /* Check if overwriting previous insertion with same key */
236  if (bucket->isValid() && m_cellIds[bucket->cellIndex] == cellId)
237  {
238  targetBucket = bucket;
239  break;
240  }
241 
242  /* Tag first free slot */
243  if (!bucket->isValid() && !targetBucket)
244  targetBucket = bucket;
245 
246  /* If we have found a free bucket we can stop iterating when we are sure that the key was not previously inserted */
247  if (targetBucket && offset >= m_maxProbeLength)
248  break;
249 
250  /* Go to next bucket */
251  offset++;
252 
253  /* Resize and rebuild table if it takes too long to find a free slot */
254  if (offset > AGX_HASH_TABLE_MAX_PROBE_LENGTH)
255  {
256  this->rebuild((size_t)((float)m_capacity * AGX_HASH_TABLE_GROW_FACTOR));
257  this->insert(cellId, cellIndex);
258  return;
259  }
260  } while (trueVal);
261  }
262 #ifdef _MSC_VER
263 #pragma warning(pop)
264 #endif
265 
266  agxAssert(targetBucket);
267 
268  if (offset > m_maxProbeLength)
269  m_maxProbeLength = (UInt16)offset;
270 
271  targetBucket->cellIndex = cellIndex;
272 
273  /* Check if inserting a new value rather than overwriting an old entry */
274  if (!targetBucket->isValid())
275  {
276  m_size++;
277  targetBucket->state = Bucket::VALID;
278  }
279 
280  }

282  void erase(const CellIdT& cellId)
283  {
284  UInt32 bucketIndex = findBucket(cellId);
285  agxAssert(bucketIndex != InvalidIndex);
286 
287  Bucket *bucket = &m_buckets[bucketIndex];

316  AGX_FORCE_INLINE UInt32 findBucket(const CellIdT& cellId) const
317  {
318  agxAssert(m_buckets);
319 
320  UInt32 hashValue = m_hashFunctor(cellId);
321  size_t index = hashValue % m_capacity;
322  const Bucket *bucket = &m_buckets[index];
323 
324  /* First test is outside for-loop, hopefully getting better branch prediction */
325  if (bucket->isValid() && m_cellIds[bucket->cellIndex] == cellId)
326  return (UInt32)index;
327 
328 
329  for (size_t offset = 1; bucket->isActive() && offset <= m_maxProbeLength; offset++)
330  {
331  index = (hashValue + offset * offset) % m_capacity;
332  bucket = &m_buckets[index];
333 
334  if (bucket->isValid() && m_cellIds[bucket->cellIndex] == cellId)
335  return (UInt32)index;
336  }
337 
338  /* Not found */
339  return InvalidIndex;
340  }

342  void rebuild(size_t size)
343  {
344  size = std::max( size, size_t(AGX_HASH_TABLE_MIN_SIZE) );
345  // std::cout << "HashTable: Rebuilding and resizing from " << m_capacity << " to " << agx::nextPrime((int)(size)) << " buckets." << std::endl;
346  agxAssert(size >= m_size);
347  size_t oldNumBuckets = m_capacity;
348  Bucket *oldBuckets = m_buckets;
349 
350  m_capacity = agx::nextPrime((int)size);
351  m_buckets = (Bucket *)m_allocator.allocateBytes(m_capacity * sizeof(Bucket));
352  for (size_t i = 0; i < m_capacity; ++i)
353  ::new (&m_buckets[i]) Bucket();
354 
355  m_size = 0;
356  m_maxProbeLength = 0;
357 
358  for (size_t i = 0; i < oldNumBuckets; i++)
359  {
360  Bucket& bucket = oldBuckets[i];
361 
362  if (bucket.isValid())
363  this->insert(m_cellIds[bucket.cellIndex], bucket.cellIndex);
364  }
365 
366  m_allocator.deallocateBytes((void *)oldBuckets, oldNumBuckets * sizeof(Bucket));
367  }
