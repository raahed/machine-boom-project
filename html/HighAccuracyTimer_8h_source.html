209  AGX_FORCE_INLINE Real64 HighAccuracyTimer::getTime() const
210  {

213  {
214  std::high_resolution_clock::time_point now = std::high_resolution_clock::now();
215  std::duration<Real64> duration = duration_cast<std::duration<Real64>>(now - m_start);
216  std::duration<Real64> total = m_total + duration;
217  return total.count() * Real64(1000);

219  else
220  {
221  return m_total.count() * Real64(1000);;
222  }
223 
224 #else

240  AGX_FORCE_INLINE UInt64 HighAccuracyTimer::getCurrentTick()
241  {
242 #if defined(_MSC_VER)
243  const UInt64 retVal = getCurrentTickWindows();
244  return retVal;
245 #elif defined(AGX_APPLE_IOS) || defined(__APPLE__)
246  return mach_absolute_time();
247 #elif defined (__GNUC__) && ! defined(__INTEL_COMPILER) && defined(__ia64__)
248  unsigned long long int val;
249  __asm__ __volatile__("mov %0=ar.itc" : "=r"(val) :: "memory");
250  return(val);
251 #elif defined(__INTEL_COMPILER) && defined(__ia64__) && !defined(HAVE_TICK_COUNTER)
252  return __getReg(_IA64_REG_AR_ITC);
253 #elif defined(__ECC) && defined(__ia64__) && !defined(HAVE_TICK_COUNTER)
254  return __getReg(_IA64_REG_AR_ITC);
255 #elif (defined(__GNUC__) || defined(__ICC)) && (defined(__i386__) || defined(__x86_64__)) && !defined(HAVE_TICK_COUNTER)
256 
257  // "The CPUID instruction serializes the processor pipeline so that all of the preceding
258  // instructions must retire before it begins execution. ... This is thought to provide
259  //a more accurate cycle count on the code being measured." [intel person at forum on intel.com]
260 
261  // cpuid writes to eax,ebx,ecx, edx, tell g++ about registers being clobbered
262  // in 64bits ebx is 32bits of the larger rbx register so should not require
263  // any ifdef's in 64bit
264  // would be better to use pushad, cpuid, popad but for some unknownn reason gcc
265  // doesn't recognize pushad/popad
266  // Update: gcc has different name for pushad/popad, l postfix instead of d,
267  // but those instructions doesn't work in 64bit
268 
269 #ifdef __LP64__
270  unsigned long long int lowbits, highbits;
271  __asm__ __volatile__("rdtsc": "=a" (lowbits), "=d"(highbits) : : "ebx", "ecx");
272 #else
273  unsigned long int lowbits, highbits;
274  __asm__ __volatile__("pushal\n\t" "cpuid\n\t" "popal\n\t" "rdtsc": "=a" (lowbits), "=d"(highbits));
275  //__asm__ __volatile__("push %%ebx\n" "push %%ecx\n" "cpuid\n" "rdtsc\n" "pop %%ebx\n" "pop %%ecx": "=a" (lowbits), "=d"(highbits));
276 #endif
277 
278  return (((unsigned long long int)highbits << 32) | (unsigned long long int)lowbits);
279 
280 #else
281  agxAbort1("Timer::getCurrentTick failed!");
282  return -1;
283 #endif
284
